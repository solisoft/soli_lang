    new(value: Int, unit: String) {

                

                    130
                    ✗

                    131
                    ✗
                    
        if (unit == "days") {

                

                    132
                    ✗

                    133
                    ✗
                    
        } elsif (unit == "hours") {

                

                    134
                    ✗
                    
            multiplier = 3600.0;

                

                    135
                    ✗
                    
        } elsif (unit == "minutes") {

                

                    136
                    ✗
                    
            multiplier = 60.0;

                

                    137
                    ✗
                    
        } elsif (unit == "seconds") {

                

                    138
                    ✗
                    
            multiplier = 1.0;

                

                    139
                    ✗
                    
        } elsif (unit == "milliseconds") {

                

                    140
                    ✗
                    
            multiplier = 0.001;

                

                    141
                    ✗

                    142
                    ✗

                    144

            return;
